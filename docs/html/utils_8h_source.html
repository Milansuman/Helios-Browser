34 template <class T>
35 std::vector<std::basic_string<T>> split(const std::basic_string<T> &s,
36 const std::basic_string<T> &delimiter) {
37 std::vector<std::basic_string<T>> tokens;
38 typename std::basic_string<T>::size_type start = 0;
39 typename std::basic_string<T>::size_type end = s.find(delimiter);
40 while (end != std::basic_string<T>::npos) {
41 tokens.push_back(s.substr(start, end - start));
42 start = end + delimiter.size();
43 end = s.find(delimiter, start);
44 }
45 tokens.push_back(s.substr(start));
46 return tokens;
47 }

51 auto start = str.begin();
52 while (start != str.end() && std::isspace(*start)) {
53 start++;
54 }
55
56 auto end = str.end();
57 do {
58 end--;
59 } while (std::distance(start, end) > 0 && std::isspace(*end));
60
61 return {start, end + 1};
62 }
